class Solution {
public:
    vector<int> createTargetArray(vector<int>& nums, vector<int>& index) {
        vector<int> ans;
        for(int i=0;i<index.size();i++)
        {
            int ind=index[i];
            ans.insert(ans.begin()+ind,nums[i]);
        }
        return ans;
    }
};
